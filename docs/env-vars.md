# Environment Variables

아래 값은 예시이며, 실제 값은 인프라 환경에 맞게 조정하세요. 예시는 `application.properties`에 매핑되는 환경 변수 기준입니다.

```properties
# App
SPRING_APPLICATION_NAME=healthcareproject-backend
SERVER_PORT=8080
APP_VERSION=1.0.0
APP_CORS_ALLOWED_ORIGINS=https://your-frontend.example

# Frontend URLs
APP_FRONTEND_RESET_PASSWORD_URL=https://your-frontend.example/reset-password
APP_RESET_PASSWORD_COOLDOWN_SECONDS=60

# Datasource (PostgreSQL)
SPRING_DATASOURCE_URL=jdbc:postgresql://<HOST>:<PORT>/<DB_NAME>
SPRING_DATASOURCE_DRIVER_CLASS_NAME=org.postgresql.Driver
SPRING_DATASOURCE_USERNAME=<USERNAME>
SPRING_DATASOURCE_PASSWORD=<PASSWORD>

# JPA (optional)
SPRING_JPA_HIBERNATE_DDL_AUTO=validate
SPRING_JPA_PROPERTIES_HIBERNATE_DIALECT=org.hibernate.dialect.PostgreSQLDialect
SPRING_JPA_OPEN_IN_VIEW=false
SPRING_JPA_SHOW_SQL=false
SPRING_JPA_PROPERTIES_HIBERNATE_FORMAT_SQL=false

# Flyway (prod)
SPRING_FLYWAY_ENABLED=true
SPRING_FLYWAY_LOCATIONS=classpath:db/migration
SPRING_FLYWAY_BASELINE_ON_MIGRATE=true
SPRING_FLYWAY_BASELINE_VERSION=1

# Redis
SPRING_DATA_REDIS_HOST=<REDIS_HOST>
SPRING_DATA_REDIS_PORT=6379

# JWT
APP_JWT_SECRET=<LONG_RANDOM_SECRET_32CHARS_MINIMUM>
APP_JWT_ACCESS_TOKEN_VALIDITY_SECONDS=3600
APP_JWT_REFRESH_TOKEN_VALIDITY_SECONDS=1209600

# Mail (SMTP)
SMTP_MAIL_HOST=<SMTP_HOST>
SMTP_MAIL_PORT=587
SMTP_MAIL_USERNAME=<SMTP_USERNAME>
SMTP_MAIL_PASSWORD=<SMTP_PASSWORD>
SMTP_MAIL_FROM_ADDRESS=no-reply@example.com
SMTP_MAIL_FROM_NAME=Healthcare Project

# Email verification / reset (Redis)
APP_VERIFICATION_EMAIL_TTL_SECONDS=300
APP_VERIFICATION_EMAIL_PREFIX=verify:email:
APP_VERIFICATION_COOLDOWN_SECONDS=60
APP_VERIFICATION_COOLDOWN_PREFIX=verify:cooldown:
APP_RESET_PASSWORD_TTL_SECONDS=900
APP_RESET_PASSWORD_PREFIX=reset:pw:

# Social OAuth
SOCIAL_GOOGLE_USERINFO_URL=https://openidconnect.googleapis.com/v1/userinfo
SOCIAL_GOOGLE_CLIENT_ID=<GOOGLE_CLIENT_ID>
SOCIAL_GOOGLE_CLIENT_SECRET=<GOOGLE_CLIENT_SECRET>
SOCIAL_GOOGLE_TOKEN_URL=https://oauth2.googleapis.com/token

SOCIAL_KAKAO_ME_URL=https://kapi.kakao.com/v2/user/me
SOCIAL_KAKAO_CLIENT_ID=<KAKAO_CLIENT_ID>
SOCIAL_KAKAO_CLIENT_SECRET=<KAKAO_CLIENT_SECRET>
SOCIAL_KAKAO_TOKEN_URL=https://kauth.kakao.com/oauth/token

SOCIAL_NAVER_ME_URL=https://openapi.naver.com/v1/nid/me
SOCIAL_NAVER_CLIENT_ID=<NAVER_CLIENT_ID>
SOCIAL_NAVER_CLIENT_SECRET=<NAVER_CLIENT_SECRET>
SOCIAL_NAVER_TOKEN_URL=https://nid.naver.com/oauth2.0/token

SOCIAL_HTTP_CONNECT_TIMEOUT_MS=3000
SOCIAL_HTTP_READ_TIMEOUT_MS=5000

# AI (OpenAI)
OPENAI_API_KEY=<OPENAI_API_KEY>
OPENAI_MODEL=gpt-4.1-mini
OPENAI_TEMPERATURE=0.2

# AWS S3
AWS_S3_BUCKET=<S3_BUCKET_NAME>
AWS_REGION=<AWS_REGION>
AWS_ACCESS_KEY_ID=<AWS_ACCESS_KEY_ID>
AWS_SECRET_ACCESS_KEY=<AWS_SECRET_ACCESS_KEY>
```
